VAR SIZE, RED, GREEN, BLUE 

SIZE := 10
RED := 10
GREEN := 0
BLUE := 0

REPEAT 30 [
  % MAKING A RECTANGLE
	REPEAT 4 [
		FD SIZE
		RT 90
	]
  % DESPLACE A FEW GRADIANS
	RT 10

  % CHAGING COLOR
	COLOR RED GREEN BLUE
	SIZE := (SIZE + 5)
	RED := ((RED + 10) MOD 255)

  IFELSE SIZE MOD 2 = 0 [PENUP] [PENDOWN]
]
