VAR SIZE, RED, GREEN, BLUE, I, J, ANGLE

SIZE := 10
RED := 10
ANGLE := 10
GREEN := 128
BLUE := 240
PENWIDTH(3)

I := 1
WHILE (I < 10)  [
	J := 1
	WHILE (J <= 6) [
		FD (SIZE)
		RT (60)
		J := J + 1
	]
	RT (ANGLE)

	RED := ((RED + 10) MOD 255)
	GREEN := (GREEN + 20) MOD 255
	BLUE := (BLUE + 40) MOD 255

	COLOR (RED, GREEN, BLUE)
	SIZE := (SIZE + 10)
	ANGLE := ANGLE + 5

	IFELSE (I MOD 2 = 0) [PENUP()] [PENDOWN()]
	
	I := I + 1
]
